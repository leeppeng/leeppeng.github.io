Js-Promise


promise是用来干什么的？
用来处理回调，避免回调地狱
用来更好地处理异步行为

Promise 有三个状态
-pending 初始状态 ，未定
-fulfilled 完成状态 ，成功 （resolve => fulfilled）
-reject 失败

.then() 有两个参数，这两个参数都是函数，它们其实是在.then()方法里进行的回调，成功：resolve失败：reject
.then 调用的是一个Promise对象，每一个.then()方法，都会获得上一次.then()的执行结果

`
// 创建了一个Peomise对象实例 ，_promise1
let _promise1 = new Promise(function(resolve, reject){
setTimeout(function(){
resolve(‘2秒后，打印一个xx’)
},2000)
})

_promise1
.then(function(_data){
// 第一个.then()，就是第一个嵌套函数
// resolve,执行它
// 因为在seTimeout里调用的就是resolve
console.log(_data) //2秒后，打印一个xx
return ‘lee’
},function(_err){
//reject
})
.then(function(data){
// 第二个.then()，就是第二个嵌套函数
// 它执行的data是第一个.then()返回的值
console.log(data) // lee
return ${data} peng
},function(err){
//reject
})
.then(function(data){
// 第三个.then()，就是第三个嵌套函数
// 它执行的data是第二个.then()返回的值
console.log(data) // lee peng
},function(err){
//reject
})
`

.catch()
也返回一个Promise对象，主要用于捕获异常，所以一般.then()的第二个参数都不写，都用.catch()方法

// 创建了一个Peomise对象实例 ，_promise2 let _promise2 = new Promise(function(resolve, reject){ setTimeout(function(){ reject('这是一个失败操作') },2000) }) _promise2 .then(function(){ console.log(成功会走这) },function(_err){ // 如果这里有了错误捕获，.catch()就不起作用了 console.log(_err: ${_err}) }) .catch(function(err){ console.log(catch: ${err}`)
})

`

.all()方法
接收一个 promise 对象的数组作为参数，当这个数组里的所有 promise 对象全部变为resolve或 有 reject 状态出现的时候，它才会去调用 .then 方法,它们是并发执行的。
它就是迭代数组，用来操作一个异步什么的，其实就是一个循环
.all() 是一个静态方法，可以直接调用，不用new

`
{
let p1 = new Promise((reslove, reject) =>{
setTimeout(reslove, 1000, ‘111’)
})
let p2 = new Promise((reslove, reject) =>{
setTimeout(reject, 2000, ‘222’)
})
let p3 = new Promise((reslove, reject) =>{
setTimeout(reslove, 3000, ‘333’)
})

Promise.all([p1,p2,p3])
.then(result => {
console.log(result)
})
.catch(err=>{
console.log(err)
})
}
//Promse.all在处理多个异步处理时非常有用，比如说一个页面上需要等两个或多个ajax的数据回来以后才正常显示，在此之前只显示loading图标。
// Promise.all获得的成功结果的数组里面的数据顺序和Promise.all接收到的数组顺序是一致的，即p1的结果在前，即便p1的结果获取的比p2要晚。这带来了一个绝大的好处：在前端开发请求数据的过程中，偶尔会遇到发送多个请求并根据请求顺序获取和使用数据的场景，使用Promise.all毫无疑问可以解决这个问题。
//代码模拟
let wake = (time) =>{
return new Promise((resolve, reject)=>{
setTimeout(resolve, time, ${time/1000} 秒后醒来)
})
}
let p1 = wake(2000)
let p2 = wake(3000)
Promise.all([p1,p2])
.then((result) => {
console.log(result)
})
.catch((err) => {
console.log(err)
})
`

.race()
它的参数是一组可以迭代的参数
只要一组参数中，有任意一个发生变化，那么Pormise对象的状态就改变了
Promise.race([p1, p2, p3])里面哪个结果获得的快，就返回那个结果，不管结果本身是成功状态还是失败状态

`
let p1 = new Promise(function(resolve, reject){
setTimeout(resolve, 300, ‘p1…XXX’)
})
let p2 = new Promise(function(resolve, reject){
setTimeout(resolve, 30, ‘P2….XXXX’)
})
let p3 = new Promise(function(resolve, reject){
setTimeout(reject,200,’P3…XXX’)
})

Promise
.race([p1,p2,p3])
.then(function(data){
console.log(data) // 谁先变化就用谁的值–p2最先完成 P2….XXXX
})
.catch(function(err){
console.log(err)
})
`

resolve() 接收一个参数
–可以是普通的值
–可以是一个含的.then()方法的对象
–可以是一个Promise实例
三种形式的参数，无非是对数据的三种包装而已

`
//（1）参数是一个 Promise 实例
//如果参数是 Promise 实例，那么Promise.resolve将不做任何修改、原封不动地返回这个实例。

//（2）参数是一个thenable对象
//thenable对象指的是具有then方法的对象，比如下面这个对象。
let thenable = {
then: function(resolve, reject) {
resolve(42);
}
};
// Promise.resolve方法会将这个对象转为 Promise 对象，然后就立即执行thenable对象的then方法。
let p1 = Promise.resolve(thenable);
p1.then(function(value) {
console.log(value); // 42
});
//上面代码中，thenable对象的then方法执行后，对象p1的状态就变为resolved，从而立即执行最后那个then方法指定的回调函数，输出 42。

//（3）参数不是具有then方法的对象，或根本就不是对象
//如果参数是一个原始值，或者是一个不具有then方法的对象，则Promise.resolve方法返回一个新的 Promise 对象，状态为resolved。
const p = Promise.resolve(‘Hello’);
p.then(function (s){
console.log(s)// Hello
});
// 上面代码生成一个新的 Promise 对象的实例p。
// 由于字符串Hello不属于异步操作（判断方法是字符串对象不具有 then 方法），返回 Promise 实例的状态从一生成就是resolved，所以回调函数会立即执行。
// Promise.resolve方法的参数，会同时传给回调函数。

`

reject()
它接收的参数就是异常，此时返回Promise对象的状态是reject –失败
let L1 = Promise.reject('失败的操作') L1.then(function(data){ // 成功的 }) .catch(function(err){ console.log(catch${err}) })
